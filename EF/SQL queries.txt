#Select every real workhour that doesnt belong to the timeshif(ex wrong month)
select 
	r.Id,r.StartOn,r.TimeShiftId ,r.EmployeeId,
	t.Month 
from RealWorkHours r 
inner join TimeShifts t on r.TimeShiftId= t.Id 
where t.Month=1 and MONTH(r.StartOn)=2
